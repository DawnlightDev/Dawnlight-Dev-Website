Keyboard Instrument

For this project, we create a keyboard instrument that plays 4 notes while also lighting up an LED light when the key is pressed.

Parts Required For Assembly

  • Power Cables
  • 1 Piezo
  • 3 Coloured LED Lights
  • 4 10kohm Resistors
  • 1 220 Ohm Resistor
  • 1 Megohm Resistor
  • 4 Switches
  • 4 Small Jumper Wires
  • 4 Medium/Large Jumper Wire


        int notes[] = {262,294,330,349};

        void setup() {
          Serial.begin(9600);
          pinMode(7, OUTPUT);
          pinMode(11, OUTPUT);
          pinMode(10, OUTPUT);
        }
        
        void loop() {
          int keyVal = analogRead(A0);
          Serial.println(keyVal);
          digitalWrite(11, LOW);
          digitalWrite(7, LOW);
          digitalWrite(10, LOW);
          
          if(keyVal > 1010){
            tone(8, notes[0]);
            digitalWrite(7, HIGH); 
          }
        
          else if(keyVal >= 990 && keyVal <= 1010){
            tone(8, notes[1]);
            digitalWrite(11, HIGH);
          }
          
          else if(keyVal >= 505 && keyVal <= 515){
            tone(8, notes[2]);
            digitalWrite(11, HIGH);
          }
        
          else if(keyVal >= 5 && keyVal <= 300){
            tone(8, notes[3]);
            digitalWrite(10, HIGH); 
          }
        
          else{
            noTone(8);
          }
        }
